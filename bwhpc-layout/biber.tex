    \iffieldundef{#3}

      {\bibstring{\thefield{#1}}}

    \iffieldundef{#3}
      {}
      {\mkdayzeros{\thefield{#3}}\adddot

       \iffieldundef{#1}
     {}
     {\iffieldundef{#3}{/}{\adddot\thinspace}}}%
    \iffieldbibstring{#1}
